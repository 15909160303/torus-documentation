import Tabs from "@theme/Tabs";

Install OpenLogin SDK using either
[npm](https://www.npmjs.com/package/@toruslabs/torus-openlogin) or CDN:

<Tabs
  defaultValue="npm"
  values={[
    { label: "npm", value: "npm" },
    { label: "jsdelivr", value: "jsdelivr" },
  ]}
>
<TabItem value="npm">

```shell
npm i --save @toruslabs/torus-openlogin
```

</TabItem>

<TabItem value="jsdelivr">

```html
<script src="https://cdn.jsdelivr.net/npm/@toruslabs/openlogin@0.2.0/dist/openlogin.umd.min.js"></script>
```

</TabItem>

</Tabs>
